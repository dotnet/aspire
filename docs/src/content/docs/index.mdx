## Code-Centric Control

Define your orbit however you want. Easily add services, connect them to each other, and get everything running with a single command.

export const ts = `
import { createBuilder } from "./.modules/distributed-application";

const builder = createBuilder();

// Add frontend and API to orbit
const frontend = builder.addNpmApp("frontend", "../frontend")
    .withHttpEndpoint({ env: "PORT", targetPort: 3000 })
    .withNpmPackageInstallation();

const api = builder.addProject("api", "../api/ApiService")
    .withHttpEndpoint({ env: "API_URL" });

// Add database services to orbit
const redis = builder.addRedis("cache");
const postgres = builder.addPostgres("db").addDatabase("appdata");

// Connect services in orbit
frontend.withReference(api);
api.withReference(redis);
api.withReference(postgres);

// Add telemetry dashboard
builder.addProject("dashboard")
    .withReference(api)
    .withReference(frontend);

// Launch your services into orbit
builder.build().run();
`;
